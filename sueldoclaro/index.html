--- DISEÑO DE LA INTERFAZ GRÁFICA (Estilo Azul Claro) --- root = tk.Tk() root.title("Sueldo Claro Full Pro") root.geometry("420x680") root.configure(bg="#0056b3") # Fondo azul oscuro de base estructural # Paleta de colores aplicada a la interfaz COLOR_FONDO_AZUL = "#1a73e8" # Fondo azul claro vibrante COLOR_TEXTO_BLANCO = "#ffffff" COLOR_CUADRO_BLANCO = "#ffffff" # Contenedor principal con soporte de desplazamiento (Scroll) canvas = tk.Canvas(root, bg=COLOR_FONDO_AZUL, highlightthickness=0) scrollbar = tk.Scrollbar(root, orient="vertical", command=canvas.yview) frame = tk.Frame(canvas, bg=COLOR_FONDO_AZUL, padx=25, pady=20) frame.bind("", lambda e: canvas.configure(scrollregion=canvas.bbox("all"))) canvas.create_window((0, 0), window=frame, anchor="nw") canvas.configure(yscrollcommand=scrollbar.set) # --- ELEMENTOS VISUALES --- # Bloque Usuario y Fecha tk.Label(frame, text="👤 Nombre de Usuario:", bg=COLOR_FONDO_AZUL, fg=COLOR_TEXTO_BLANCO, font=("Arial", 10, "bold")).pack(anchor="w", pady=(5,0)) entry_usuario = tk.Entry(frame, bg=COLOR_CUADRO_BLANCO, fg="black", insertbackground="black", font=("Arial", 11)) entry_usuario.pack(fill="x", pady=5) tk.Label(frame, text="📅 Mes Correspondiente:", bg=COLOR_FONDO_AZUL, fg=COLOR_TEXTO_BLANCO, font=("Arial", 10, "bold")).pack(anchor="w") combo_mes = ttk.Combobox(frame, values=["Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio", "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"], state="readonly", font=("Arial", 10)) combo_mes.set("Junio") combo_mes.pack(fill="x", pady=5) tk.Label(frame, text="📆 Año:", bg=COLOR_FONDO_AZUL, fg=COLOR_TEXTO_BLANCO, font=("Arial", 10, "bold")).pack(anchor="w") combo_anio = ttk.Combobox(frame, values=["2025", "2026", "2027", "2028"], state="readonly", font=("Arial", 10)) combo_anio.set("2026") combo_anio.pack(fill="x", pady=5) # Línea divisoria visual tk.Frame(frame, height=2, bg="#ffffff").pack(fill="x", pady=15) # Formulario de Carga Dinámica con Emojis campos_config = [ ("💰 Sueldo Base ($):", "entry_sueldo"), ("⚡ Cantidad Horas Extra:", "entry_horas"), ("🏠 Gasto Alquiler ($):", "entry_alquiler"), ("💳 Gasto Tarjetas ($):", "entry_tarjeta"), ("🏛️ Gasto Préstamos ($):", "entry_prestamo"), ("🛒 Otros Gastos / Varios ($):", "entry_varios") ] entries_dict = {} for label_txt, var_name in campos_config: tk.Label(frame, text=label_txt, bg=COLOR_FONDO_AZUL, fg=COLOR_TEXTO_BLANCO, font=("Arial", 10, "bold")).pack(anchor="w", pady=(4,0)) # Cuadros blancos limpios con texto negro legible e = tk.Entry(frame, bg=COLOR_CUADRO_BLANCO, fg="black", insertbackground="black", font=("Arial", 11)) e.pack(fill="x", pady=4) entries_dict[var_name] = e # Mapeo de variables individuales para mantener compatibilidad entry_sueldo = entries_dict["entry_sueldo"] entry_horas = entries_dict["entry_horas"] entry_alquiler = entries_dict["entry_alquiler"] entry_tarjeta = entries_dict["entry_tarjeta"] entry_prestamo = entries_dict["entry_prestamo"] entry_varios = entries_dict["entry_varios"] # Botón de Acción Principal (Verde WhatsApp Estilo Fluido) btn_enviar = tk.Button( frame, text="🚀 ENVIAR TICKET A WHATSAPP", bg="#25D366", fg="white", font=("Arial", 11, "bold"), activebackground="#20ba5a", activeforeground="white", command=calcular_y_enviar, relief="flat", pady=8 ) btn_enviar.pack(pady=25, fill="x") # Carga inicial del estado guardado cargar_datos() # Empaquetado final del sistema de scroll canvas.pack(side="left", fill="both", expand=True) scrollbar.pack(side="right", fill="y") root.mainloop()
